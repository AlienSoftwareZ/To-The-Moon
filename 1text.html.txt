<script>

let money=[10,30,70,150]

let selectedIndex=null
let currentPattern=""
let resultStatus=[]

let profit=0
let peak=0
let maxDD=0
let loseStreak=0
let winCount=0
let loseCount=0

let historyProfit=[]
let round=0

function generatePattern(){

round++
document.getElementById("round").innerText=round

currentPattern=""
resultStatus=[]
selectedIndex=null

for(let i=0;i<6;i++){

let val=Math.random()<0.5?"B":"P"

currentPattern+=val
resultStatus.push(null)

}

renderPattern()

}

function renderPattern(){

let box=document.getElementById("patternBox")
box.innerHTML=""

for(let i=0;i<currentPattern.length;i++){

let val=currentPattern[i]

let div=document.createElement("div")
div.className="circle "+val

if(i===selectedIndex) div.classList.add("selected")

if(resultStatus[i]=="win") div.classList.add("win")
if(resultStatus[i]=="lose") div.classList.add("lose")

div.innerText=val

div.onclick=function(){

if(resultStatus[i]!=null) return

selectedIndex=i
renderPattern()

}

box.appendChild(div)

}

}

function resultBet(win){

if(selectedIndex===null){
alert("เลือกไม้ก่อน")
return
}

let stepIndex=Math.min(selectedIndex,money.length-1)

let bet=money[stepIndex]

let totalLose=0
for(let i=0;i<stepIndex;i++){
totalLose+=money[i]
}

let net=0

if(win){

net=bet-totalLose
profit+=net

loseStreak=0
winCount++

resultStatus[selectedIndex]="win"

}else{

net=-bet
profit+=net

loseStreak++
loseCount++

resultStatus[selectedIndex]="lose"

}

saveHistory(win,net,stepIndex)

selectedIndex=null

if(profit>peak) peak=profit

let dd=peak-profit
if(dd>maxDD) maxDD=dd

historyProfit.push(profit)

updateUI()
updateChart()
renderPattern()

}

function saveHistory(win,net,stepIndex){

let box=document.getElementById("historyBox")

let patternHTML=""

for(let p of currentPattern){

patternHTML+=`<div class="smallCircle ${p==="B"?"smallB":"smallP"}">${p}</div>`

}

let resultClass=win?"resultWin":"resultLose"
let netClass=net>=0?"netWin":"netLose"

let card=document.createElement("div")

card.className="historyCard"

card.innerHTML=`

<div style="display:flex;justify-content:space-between">
<div>รอบ ${round}</div>
<div>ไม้ ${stepIndex+1}</div>
</div>

<div style="display:flex;gap:6px;margin:6px 0">
${patternHTML}
</div>

<div style="display:flex;justify-content:space-between">
<div class="${resultClass}">${win?"WIN":"LOSE"}</div>
<div class="${netClass}">${net>0?"+":""}${net}</div>
</div>
`

box.prepend(card)

}

function updateUI(){

document.getElementById("profit").innerText=profit
document.getElementById("profit").style.color=profit>=0?"#00e676":"#ff5252"

document.getElementById("maxDD").innerText=maxDD
document.getElementById("loseStreak").innerText=loseStreak

let total=winCount+loseCount
let wr=total?((winCount/total)*100).toFixed(1):0

document.getElementById("winrate").innerText=wr+"%"

}

function resetAll(){
location.reload()
}

let chart=new Chart(document.getElementById("profitChart"),{

type:"line",

data:{
labels:[],
datasets:[{
data:[],
borderColor:"#d4af37",
tension:.35
}]
},

options:{
plugins:{legend:{display:false}}
}

})

function updateChart(){

chart.data.labels=historyProfit.map((_,i)=>i+1)
chart.data.datasets[0].data=historyProfit
chart.update()

}
